def sum_d(N):
    total = 0
    while N > 0:
        d = N % 10
        total += d
        N //= 10
    return total

t = int(input())
for _ in range(t):
    N = int(input())
    print(sum_d(N))

# problem5 : Write a function to check whether a number is prime.
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

n = int(input())
print("Prime" if is_prime(n) else "Not Prime")
```

# Problem 6 : Write a function to reverse a number
def reverse_num(N):
    rev = 0
    while N > 0:
        rev = rev * 10 + N % 10
        N //= 10
    return rev

N = int(input())
print(reverse_num(N))
```

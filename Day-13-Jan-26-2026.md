```python
#Problem 6 : Possible or Not
# cook your dish here

import sys
input = sys.stdin.readline

T = int(input())
for _ in range(T):
    N, B = map(int, input().split())
    A = list(map(int, input().split()))
    
    candidates = []
    for x in A:
        if (x & B) == B:
            candidates.append(x)
    
    if not candidates:
        print("NO")
        continue
    
    ans = candidates[0]
    for x in candidates[1:]:
        ans &= x
    
    if ans == B:
        print("YES")
    else:
        print("NO")

```

```python
#Problem 7 : Factorial
# cook your dish here
import sys
input = sys.stdin.readline

def trailing_zeros(n):
    if n < 5:          # base case
        return 0
    return n // 5 + trailing_zeros(n // 5)

T = int(input())
for _ in range(T):
    N = int(input())
    print(trailing_zeros(N))

```

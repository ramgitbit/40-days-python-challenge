# Priblem 1: Write a function count_vowels(s) that returns the number of vowels (a, e, i, o, u) in the given string s. 

#solution:
def count_vowels(s):
    count = 0
    for ch in s.lower():
        if ch in 'aeiou':
            count += 1
    return count

s = input()
print(count_vowels(s))

#problem 2 : Write a function factorial(n) that returns the factorial of a number n.

#solution:
 def factorial(n):
    fact = 1
    for i in range(1, n + 1):
        fact *= i
    return fact

n = int(input())
print(factorial(n))

# problem 3 : Write a function is_prime(n) that returns "Prime" if the number is prime, otherwise "Not Prime".

#solution:
def is_prime(n):
    if n <= 1:
        return "Not Prime"
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return "Not Prime"
    return "Prime"

n = int(input())
print(is_prime(n))

# problem 4: 
Write a function sum_of_digits(n) that returns the sum of digits of a number.

#solution:
def sum_of_digits(n):
    total = 0
    while n > 0:
        total += n % 10
        n //= 10
    return total

n = int(input())
print(sum_of_digits(n))

# Priblem 1: Given a list of numbers, remove all duplicate values without using set, keep the original order, and then reverse the list.

#solution:
nums = [1, 2, 3, 2, 4, 1, 5]

unique = []
for x in nums:
    if x not in unique:
        unique.append(x)

unique.reverse()
print(unique)

# problem 3 : Given a dictionary of students and their marks, increase the marks of every student who scored less than 40 by 5 marks, but do not let any score exceed 40.

#solution:
marks = {"Aman": 35, "Riya": 42, "Kunal": 38}

for name, score in marks.items():
    if score < 40:
        marks[name] = min(score + 5, 40)

print(marks)

Calculate and print the average marks of each student.

    "Riya": [88, 79, 85],
    "Kunal": [60, 65, 70]

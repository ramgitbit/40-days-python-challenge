```python
#Problem 6 : Possible or Not
# cook your dish here

import sys
input = sys.stdin.readline

T = int(input())
for _ in range(T):
    N, B = map(int, input().split())
    A = list(map(int, input().split()))
    
    candidates = []
    for x in A:
        if (x & B) == B:
            candidates.append(x)
    
    if not candidates:
        print("NO")
        continue
    
    ans = candidates[0]
    for x in candidates[1:]:
        ans &= x
    
    if ans == B:
        print("YES")
    else:
        print("NO")

```

```python
#Problem 7 : Factorial
# cook your dish here
import sys
input = sys.stdin.readline

def trailing_zeros(n):
    if n < 5:          # base case
        return 0
    return n // 5 + trailing_zeros(n // 5)

T = int(input())
for _ in range(T):
    N = int(input())
    print(trailing_zeros(N))

```

```python
#Problem 6 : Trimality Test
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

T = int(input())
for _ in range(T):
    N = int(input())
    if is_prime(N):
        print("yes")
    else:
        print("no")
```

```python
#Problem 8 : Sum of Natural Numbers
class Solution:
    def findSum(self, n):
    
        if n > 0:
            return n + self.findSum(n-1)
        else:
            return 0
```

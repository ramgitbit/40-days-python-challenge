# Priblem 1: Given a list of numbers, remove all duplicate values without using set, keep the original order, and then reverse the list.

#solution:
nums = [1, 2, 3, 2, 4, 1, 5]

unique = []
for x in nums:
    if x not in unique:
        unique.append(x)

unique.reverse()
print(unique)

# problem 3 : Given a dictionary of students and their marks, increase the marks of every student who scored less than 40 by 5 marks, but do not let any score exceed 40.

#solution:
marks = {"Aman": 35, "Riya": 42, "Kunal": 38}

for name, score in marks.items():
    if score < 40:
        marks[name] = min(score + 5, 40)

def sum_d(N):
    total = 0
    while N > 0:
        d = N % 10
        total += d
        N //= 10
    return total

t = int(input())

    N = int(input())

# problem5 : Write a function to check whether a number is prime.
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

n = int(input())
print("Prime" if is_prime(n) else "Not Prime")

# Problem 6 : Write a function to reverse a number
def reverse_num(N):
    rev = 0
    while N > 0:
        rev = rev * 10 + N % 10
        N //= 10
    return rev

N = int(input())
print(reverse_num(N))

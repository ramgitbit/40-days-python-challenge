```python
# Priblem 2 : Sum of Digits
# cook your dish here
t = int(input())
for _ in range(t):
    n = int(input())
    sum = 0
    while n != 0:
        d = n % 10
        sum += d
        n //= 10
    print(sum)
```

```python
# Priblem 5 : first and last  Digits
# cook your dish here
t = int(input())
for _ in range(t):
    n = int(input())
    last_digit = n % 10
    first_digit = n
    while first_digit >= 10:
        first_digit //= 10
    print(first_digit + last_digit)
```

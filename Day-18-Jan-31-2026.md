```python
#Problem :4 Profit Increament
# cook your dish here
t = int(input())
for _ in range(t):
    x, y = map(int, input().split())
    if x % 100 ==0 :
        p = (x+((x*10)/100)) - (x-y)   # profit (y)= sellimg - buying : buying == selling - profit : 100 -10== 90
        print(int(p)) 

#method 2 :  solving without if and else
t = int(input())
for _ in range(t):
    selling_price, profit = map(int, input().split())
    new_profit = profit + selling_price // 10
    print(new_profit)
```

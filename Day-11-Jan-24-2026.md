```python
#Problem 6 : Trimality Test
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

T = int(input())
for _ in range(T):
    N = int(input())
    if is_prime(N):
        print("yes")
    else:
        print("no")
```

```python
#Problem 8 : Sum of Natural Numbers
class Solution:
    def findSum(self, n):
    
        if n > 0:
            return n + self.findSum(n-1)
        else:
            return 0
```
